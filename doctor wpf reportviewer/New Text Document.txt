CREATE VIEW [dbo].[summary]
AS
select distinct  pid,vdate as dd from visit union select distinct pid ,pdate as dd from pdrug union select distinct pid,tdate as dd from ptest

GO

create view [dbo].[summary2]
as
select t1.pid, t2.name,t2.Gender,YEAR(GETDATE()) - t2.bdate AS Age,blood,mobile,t1.dd as vdate from summary t1,patient t2 where t1.pid=t2.code

SET ANSI_NULLS ON
GO
SET QUOTED_IDENTIFIER ON
GO
CREATE VIEW [dbo].[drugsv]
AS
SELECT dbo.pdrug.pdrug, dbo.pdrug.drug, CONVERT(date, dbo.pdrug.pdate) AS pdate, dbo.pdrug.pid
FROM     dbo.patient INNER JOIN
                  dbo.pdrug ON dbo.patient.code = dbo.pdrug.pid

CREATE VIEW [dbo].[rep1]
AS
SELECT dbo.patient.name, dbo.patient.bdate, dbo.patient.Gender, dbo.pdrug.drug, dbo.pdrug.pdate

CREATE VIEW [dbo].[testv]
AS
SELECT dbo.ptest.pid, dbo.ptest.tcode, dbo.ptest.result, CONVERT(date, dbo.ptest.tdate) AS tdate, dbo.ptest.ptid, dbo.test.name

GO

CREATE VIEW [dbo].[VDrug]
AS
SELECT dbo.pdrug.pid, dbo.pdrug.drug, CONVERT(date, dbo.pdrug.pdate) AS pdate, dbo.patient.name, YEAR(GETDATE()) - dbo.patient.bdate AS Age, dbo.patient.Gender
FROM     dbo.pdrug INNER JOIN
                  dbo.patient ON dbo.pdrug.pid = dbo.patient.code

GO

SET ANSI_NULLS ON
GO
SET QUOTED_IDENTIFIER ON
GO
CREATE VIEW [dbo].[Vhistory]
AS
SELECT dbo.visit.pid, dbo.visit.vdate, dbo.visit.report, YEAR(GETDATE()) - dbo.patient.bdate AS Age, dbo.patient.Gender, dbo.patient.name
FROM     dbo.visit INNER JOIN
                  dbo.patient ON dbo.visit.pid = dbo.patient.code

GO
/****** Object:  View [dbo].[visit1]    Script Date: 2020-05-31 12:14:10 PM ******/
SET ANSI_NULLS ON
GO
SET QUOTED_IDENTIFIER ON
GO
CREATE VIEW [dbo].[visit1]
AS
SELECT CONVERT(date, pdate) AS pdate, pid
FROM     pdrug
UNION
SELECT CONVERT(date, tdate) AS pdate, pid AS pid
FROM     ptest
UNION
SELECT CONVERT(date, vdate) AS pdate, pid AS pid
FROM     visit

GO

SET ANSI_NULLS ON
GO
SET QUOTED_IDENTIFIER ON
GO
CREATE VIEW [dbo].[VTest]
AS
SELECT dbo.ptest.pid, dbo.test.name, dbo.ptest.result, CONVERT(date, dbo.ptest.tdate) AS tdate, dbo.patient.name AS pname, YEAR(GETDATE()) - dbo.patient.bdate AS Age, dbo.patient.Gender
FROM     dbo.ptest INNER JOIN
                  dbo.test ON dbo.ptest.tcode = dbo.test.code INNER JOIN
                  dbo.patient ON dbo.ptest.pid = dbo.patient.code

GO

CREATE VIEW [dbo].[VTestP]
AS
SELECT dbo.ptest.pid, dbo.test.name, dbo.ptest.tdate, dbo.patient.name AS patient, dbo.patient.Gender, YEAR(GETDATE()) - YEAR(dbo.patient.bdate) AS Age
FROM     dbo.ptest INNER JOIN
                  dbo.test ON dbo.ptest.tcode = dbo.test.code INNER JOIN
                  dbo.patient ON dbo.ptest.pid = dbo.patient.code
